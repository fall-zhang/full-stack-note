## Rand

```go
package main
import (
	"math/rand"
	"time"
)
func main() {
	Println(rand.Intn(10)) //会产生 0 到 10 之间的随机整数
  
	Println(rand.Float64()) //会产生 0 到 1 之间的随机数

	rand.Seed(time.Now().Unix()) // Seed 生成的随机数
	myrand := random(1, 20)

	Println(myrand)
}
func random(min, max int) int {
	return rand.Intn(max - min) + min
}
```
